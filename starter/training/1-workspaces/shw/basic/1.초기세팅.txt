            "emmet.syntaxProfiles": {

            "[vue]": {
                "editor.formatOnSave": true,

            // 적용할 파일 타입의 결정

                "javascript",
                "javascriptreact",
                "vue"
            ],
            // 프로젝트 eslintrc.js를 읽어들임
            "eslint.options": {
                "configFile": ".eslintrc.js"
            },
            // 저장시의 자동 포맷
            // deprecated
            "eslint.autoFixOnSave": true,

            "git.autofetch": true,
